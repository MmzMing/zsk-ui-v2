# 知识库小破站 - 国际化配置

## 一、技术选型

项目使用 **react-i18next** + **i18next** + **i18next-http-backend** 实现国际化功能。

| 技术 | 版本 | 用途 |
|------|------|------|
| i18next | ^24.x | 国际化核心库 |
| react-i18next | ^15.x | React 集成钩子 |
| i18next-http-backend | ^3.x | 异步加载翻译文件 |

## 四、配置文件

### 4.1 i18n 初始化配置

**文件**: `src/locales/index.ts`

```typescript
import i18n from 'i18next'
import { initReactI18next } from 'react-i18next'
import HttpBackend from 'i18next-http-backend'

// 导入语言包
import zhCN from './zh-CN/common.json'
import zhCNAuth from './zh-CN/auth.json'
import zhCNNavi from './zh-CN/navigation.json'

import enUS from './en-US/common.json'
import enUSAuth from './en-US/auth.json'
import enUSNavi from './en-US/navigation.json'

const resources = {
  'zh-CN': {
    common: zhCN,
    auth: zhCNAuth,
    navigation: zhCNNavi
  },
  'en-US': {
    common: enUS,
    auth: enUSAuth,
    navigation: enUSNavi
  }
}

i18n
  .use(HttpBackend)
  .use(initReactI18next)
  .init({
    resources,
    lng: localStorage.getItem('locale') || 'zh-CN',
    fallbackLng: 'zh-CN',
    debug: import.meta.env.DEV,
    
    interpolation: {
      escapeValue: false
    },
    
    // 命名空间配置
    ns: ['common', 'auth', 'navigation'],
    defaultNS: 'common',
    
    // 加载配置
    load: 'currentOnly',
    
    // 检测浏览器语言
    detection: {
      order: ['localStorage', 'navigator'],
      caches: ['localStorage']
    }
  })

export default i18n
```

## 五、Hook 封装

### 5.1 语言切换 Hook

**文件**: `src/hooks/useLocale.ts`

```typescript
import { useCallback } from 'react'
import { useTranslation } from 'react-i18next'
import { useAppStore } from '@/stores/app'

export type Locale = 'zh-CN' | 'en-US'

export const LOCALE_CONFIG = {
  'zh-CN': {
    name: '简体中文',
    flag: '🇨🇳',
    dayjsLocale: 'zh-cn'
  },
  'en-US': {
    name: 'English',
    flag: '🇺🇸',
    dayjsLocale: 'en'
  }
} as const

export function useLocale() {
  const { i18n } = useTranslation()
  const { locale, setLocale } = useAppStore()

  const changeLocale = useCallback(async (newLocale: Locale) => {
    await i18n.changeLanguage(newLocale)
    setLocale(newLocale)
    localStorage.setItem('locale', newLocale)
    
    // 更新 Day.js 语言
    const dayjsLocale = LOCALE_CONFIG[newLocale].dayjsLocale
    await import(`dayjs/locale/${dayjsLocale}`)
      .then(() => {
        const dayjs = (await import('dayjs')).default
        dayjs.locale(dayjsLocale)
      })
      .catch(console.error)
  }, [i18n, setLocale])

  const toggleLocale = useCallback(() => {
    const nextLocale = locale === 'zh-CN' ? 'en-US' : 'zh-CN'
    changeLocale(nextLocale)
  }, [locale, changeLocale])

  return {
    locale,
    changeLocale,
    toggleLocale,
    localeConfig: LOCALE_CONFIG[locale]
  }
}
```

## 六、状态管理集成

**文件**: `src/stores/app.ts`

```typescript
import { create } from 'zustand'
import { persist } from 'zustand/middleware'
import type { Locale } from '@/hooks/useLocale'

interface AppState {
  // 当前语言
  locale: Locale
  setLocale: (locale: Locale) => void
  
  // 其他应用状态...
  theme: 'light' | 'dark' | 'system'
  setTheme: (theme: 'light' | 'dark' | 'system') => void
}

export const useAppStore = create<AppState>()(
  persist(
    (set) => ({
      locale: 'zh-CN',
      setLocale: (locale) => set({ locale }),
      
      theme: 'system',
      setTheme: (theme) => set({ theme })
    }),
    {
      name: 'app-storage',
      partialize: (state) => ({ 
        locale: state.locale,
        theme: state.theme 
      })
    }
  )
)
```

## 九、语言切换组件

**文件**: `src/components/LocaleSwitcher.tsx`

```typescript
import { useTranslation } from 'react-i18next'
import { Button, Dropdown, DropdownTrigger, DropdownMenu, DropdownItem } from '@heroui/react'
import { Globe } from 'lucide-react'
import { useLocale, LOCALE_CONFIG, type Locale } from '@/hooks/useLocale'

export function LocaleSwitcher() {
  const { locale, changeLocale } = useLocale()
  const { t } = useTranslation()

  return (
    <Dropdown>
      <DropdownTrigger>
        <Button variant="light" isIconOnly>
          <Globe className="w-5 h-5" />
        </Button>
      </DropdownTrigger>
      <DropdownMenu
        aria-label="选择语言"
        selectedKeys={[locale]}
        onSelectionChange={(keys) => {
          const selected = Array.from(keys)[0] as Locale
          changeLocale(selected)
        }}
      >
        {(Object.keys(LOCALE_CONFIG) as Locale[]).map((key) => (
          <DropdownItem key={key}>
            {LOCALE_CONFIG[key].flag} {LOCALE_CONFIG[key].name}
          </DropdownItem>
        ))}
      </DropdownMenu>
    </Dropdown>
  )
}
```

## 十、路由国际化（可选）

如需 URL 中包含语言前缀（如 `/zh/home`、`/en/home`），可配置 React Router：

```typescript
import { BrowserRouter, Routes, Route, Navigate } from 'react-router-dom'
import { useLocale } from '@/hooks/useLocale'

function App() {
  const { locale } = useLocale()

  return (
    <BrowserRouter>
      <Routes>
        {/* 重定向到带语言前缀的路径 */}
        <Route path="/" element={<Navigate to={`/${locale}`} replace />} />
        
        {/* 带语言前缀的路由 */}
        <Route path="/:lang/*" element={<Layout />}>
          <Route index element={<Home />} />
          <Route path="about" element={<About />} />
          {/* 其他路由... */}
        </Route>
      </Routes>
    </BrowserRouter>
  )
}
```

## 十一、最佳实践

### 11.1 命名规范

- 使用嵌套对象组织文案，避免扁平化
- 键名采用 camelCase
- 按功能模块划分命名空间（common、auth、navigation等）

### 11.2 文案提取

- 不要在代码中硬编码任何用户可见的字符串
- 所有按钮文字、提示信息、错误消息都应通过 i18n 获取

### 11.3 复数处理

```json
{
  "items": {
    "one": "{{count}} 个项目",
    "other": "{{count}} 个项目"
  }
}
```

```typescript
const { t } = useTranslation()
t('items', { count: 5 }) // "5 个项目"
```

### 11.4 日期数字格式化

```typescript
import { useTranslation } from 'react-i18next'

function FormattedDate({ date }: { date: Date }) {
  const { i18n } = useTranslation()
  
  return (
    <span>
      {new Intl.DateTimeFormat(i18n.language).format(date)}
    </span>
  )
}
```
